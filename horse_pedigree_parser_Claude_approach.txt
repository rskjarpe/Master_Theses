                    if (currentDam && line.match(/^\d{2}\s+/)) {
                        const offspringMatch = line.match(/^(\d{2})\s+(.+?)\s+([SHV])\s+e\s+(.+)/);
                        if (offspringMatch) {
                            const year = parseInt(offspringMatch[1]);
                            const name = offspringMatch[2].trim();
                            const sex = offspringMatch[3];
                            const sirePart = offspringMatch[4];
                            
                            const earnings = parseEarnings(line);
                            const record = parseRaceRecord(line);
                            const time = parseTime(line);

                            const offspring = {
                                name,
                                year: year > 50 ? 1900 + year : 2000 + year,
                                sex,
                                sire: sirePart.split(/\s+/)[0],
                                earnings,
                                record,
                                time,
                                line: line
                            };
                            
                            damOffspring.push(offspring);
                        }
                    }
                }

                // Process final dam if exists
                if (currentDam && damOffspring.length > 0) {
                    processDamOffspring(horse, currentDam, damOffspring, damLevel - 1);

            // Parse time records
            const parseTime = (text) => {
                const timeMatch = text.match(/(\*?\(?\d+\)?\d+[.,]\d*[akml]+)/);
                return timeMatch ? timeMatch[1] : '';
            };

            // Extract sire information from pedigree structure
            const extractSireInfo = (lines, horseIndex) => {
                // Look for sire name and performance data in the lines following the horse header
                for (let i = horseIndex + 2; i < Math.min(horseIndex + 10, lines.length); i++) {
                    const line = lines[i].trim();
                    if (!line || line.includes('avk') || line.includes('BLUP')) break;
                    
                    // Check if this line contains a horse name and performance data
                    const earnings = parseEarnings(line);
                    const record = parseRaceRecord(line);
                    const time = parseTime(line);
                    
                    if (earnings > 0 && record.starts > 0) {
                        // Try to extract sire name from previous lines
                        for (let j = i - 1; j > horseIndex; j--) {
                            const prevLine = lines[j].trim();
                            if (prevLine && !prevLine.match(/\d/) && prevLine.length > 3) {
                                return {
                                    name: prevLine,
                                    earnings,
                                    record,
                                    time
                                };
                            }
                        }
                    }
                }
                return null;
            };

            // Extract dam information from pedigree structure
            const extractDamInfo = (lines, horseIndex) => {
                // Look for dam name and performance data
                for (let i = horseIndex + 10; i < lines.length; i++) {
                    const line = lines[i].trim();
                    if (!line) continue;
                    
                    // Check for dam section header
                    const damMatch = line.match(/^([A-ZÅÄÖ\s]+(?:\s*\([A-Z]{2,3}\))?)\s+f\.\s+(\d+)/);
                    if (damMatch) {
                        const damName = damMatch[1].replace(/\s*\([A-Z]{2,3}\)/, '').trim();
                        const birthYear = parseInt(damMatch[2]) > 50 ? 1900 + parseInt(damMatch[2]) : 2000 + parseInt(damMatch[2]);
                        
                        // Extract BLUP info if present
                        const blupMatch = line.match(/BLUP-talet är (\d+)/);
                        const blup = blupMatch ? blupMatch[1] : '';
                        
                        return {
                            name: damName,
                            birthYear,
                            blup,
                            earnings: 0,
                            record: { starts: 0, wins: 0, seconds: 0, thirds: 0 },
                            time: ''
                        };
                    }
                }
                return null;
            };

            // Main parsing function
            const parseHorseData = () => {
                if (!inputData.trim()) {
                    alert('Please paste the horse pedigree data first');
                    return;
                }

                const lines = inputData.split('\n').map(line => line.trim()).filter(line => line);
                const results = [];
                
                // Find main horse
                let horseIndex = -1;
                let currentHorse = null;
                
                for (let i = 0; i < lines.length; i++) {
                    const line = lines[i];
                    
                    // Main horse header pattern - look for number followed by horse name
                    if (line.match(/^\d+\s+[A-ZÅÄÖ\s]+$/)) {
                        const parts = line.split(/\s+/);
                        const pageNumber = parts[0];
                        const horseName = parts.slice(1).join(' ');
                        
                        // Get horse details from next line
                        const nextLine = i + 1 < lines.length ? lines[i + 1] : '';
                        const detailMatch = nextLine.match(/(\d{2}-\d+)\s+(.+?)\s+f\.\s+(\d{2})\s+(\d{2})\s+(\d{2})/);
                        
                        currentHorse = {
                            page: pageNumber,
                            name: horseName,
                            id: detailMatch ? detailMatch[1] : '',
                            sex: detailMatch ? detailMatch[2].trim() : '',
                            birthDate: detailMatch ? `${detailMatch[3]} ${detailMatch[4]} ${detailMatch[5]}` : '',
                            inbreeding: '',
                            sire: '',
                            sireEarnings: 0,
                            sireStarts: 0,
                            sire1st: 0,
                            sire2nd: 0,
                            sire3rd: 0,
                            sireBestTime: '',
                            sireRecord: { starts: 0, wins: 0, seconds: 0, thirds: 0 },
                            dam: '',
                            damEarnings: 0,
                            damStarts: 0,
                            dam1st: 0,
                            dam2nd: 0,
                            dam3rd: 0,
                            damBirthYear: null,
                            damBLUP: '',
                            damBestTime: '',
                            damRecord: { starts: 0, wins: 0, seconds: 0, thirds: 0 },
                            damsDam: '',
                            grandamsDam: '',
                            siblingEarnings: 0,
                            siblingStarts: 0,
                            sibling1st: 0,
                            sibling2nd: 0,
                            sibling3rd: 0,
                            siblingsStarted: 0,
                            totalSiblings: 0,
                            siblingsBorn2020: 0,
                            siblingData: [],
                            gdamEarnings: 0,
                            gdamStarts: 0,
                            gdam1st: 0,
                            gdam2nd: 0,
                            gdam3rd: 0,
                            gdamStarted: 0,
                            gdamTotal: 0,
                            gdamBorn2020: 0,
                            ggdamEarnings: 0,
                            ggdamStarts: 0,
                            ggdam1st: 0,
                            ggdam2nd: 0,
                            ggdam3rd: 0,
                            ggdamStarted: 0,
                            ggdamTotal: 0,
                            ggdamBorn2020: 0,
                            totalFamilyEarnings: 0,
                            totalFamilyStarts: 0,
                            totalFamilyWins: 0
                        };
                        
                        horseIndex = i;
                        break;
                    }
                }

                if (!currentHorse) {
                    alert('Could not find horse information in the provided data');
                    return;
                }

                // Extract sire and dam information from the pedigree layout
                // Look for the first horse name after the main horse info
                for (let i = horseIndex + 2; i < lines.length; i++) {
                    const line = lines[i].trim();
                    if (!line) continue;
                    
                    // Look for sire name (first major horse name)
                    if (!currentHorse.sire && line.match(/^[A-ZÅÄÖ][a-zA-ZÅÄÖäöå'\s]+$/)) {
                        currentHorse.sire = line;
                        
                        // Look for sire performance data in subsequent lines
                        for (let j = i + 1; j < Math.min(i + 5, lines.length); j++) {
                            const perfLine = lines[j];
                            const earnings = parseEarnings(perfLine);
                            const record = parseRaceRecord(perfLine);
                            const time = parseTime(perfLine);
                            
                            if (earnings > 0 || record.starts > 0) {
                                currentHorse.sireEarnings += earnings;
                                currentHorse.sireRecord.starts += record.starts;
                                currentHorse.sireRecord.wins += record.wins;
                                currentHorse.sireRecord.seconds += record.seconds;
                                currentHorse.sireRecord.thirds += record.thirds;
                                if (time && !currentHorse.sireBestTime) {
                                    currentHorse.sireBestTime = time;
                                }
                            }
                        }
                        break;
                    }
                }

                // Process dam family lines
                const familyLines = processFamilyLines(lines, currentHorse);
                
                // Calculate totals
                currentHorse.totalFamilyEarnings = currentHorse.sireEarnings + currentHorse.damEarnings + 
                    currentHorse.siblingEarnings + currentHorse.gdamEarnings + currentHorse.ggdamEarnings;
                currentHorse.totalFamilyStarts = currentHorse.sireRecord.starts + currentHorse.damRecord.starts +
                    currentHorse.siblingStarts + currentHorse.gdamStarts + currentHorse.ggdamStarts;
                currentHorse.totalFamilyWins = currentHorse.sireRecord.wins + currentHorse.damRecord.wins +
                    currentHorse.sibling1st + currentHorse.gdam1st + currentHorse.ggdam1st;

                results.push(currentHorse);
                setParsedData(results);
                generateCSV(results);
                setActiveTab('overview');
            };

            const processFamilyLines = (lines, horse) => {
                let currentDam = null;
                let damOffspring = [];
                let damLevel = 0; // 0 = dam, 1 = granddam, 2 = great-granddam

                for (let i = 0; i < lines.length; i++) {
                    const line = lines[i];
                    
                    // Parse dam sections - look for patterns like "DRIFTER'S DAYS (US)" or similar
                    const damSectionMatch = line.match(/^([A-ZÅÄÖ][A-ZÅÄÖ'\s]+(?:\s*\([A-Z]{2,3}\))?)\s+f\.\s+(\d+)/);
                    if (damSectionMatch) {
                        if (currentDam && damOffspring.length > 0) {
                            processDamOffspring(horse, currentDam, damOffspring, damLevel);
                        }
                        
                        const damName = damSectionMatch[1].replace(/\s*\([A-Z]{2,3}\)/, '').trim();
                        currentDam = {
                            name: damName,
                            originalName: damSectionMatch[1].trim(),
                            birthYear: parseInt(damSectionMatch[2]) > 50 ? 1900 + parseInt(damSectionMatch[2]) : 2000 + parseInt(damSectionMatch[2])
                        };
                        
                        // Set dam hierarchy
                        if (damLevel === 0 && horse.dam === '') {
                            horse.dam = damName;
                            horse.damBirthYear = currentDam.birthYear;
                            
                            // Extract BLUP info
                            const blupMatch = line.match(/BLUP-talet är (\d+)/);
                            if (blupMatch) {
                                horse.damBLUP = blupMatch[1];
                            }
                        } else if (damLevel === 1) {
                            horse.damsDam = damName;
                        } else if (damLevel === 2) {
                            horse.grandamsDam = damName;
                        }
                        
                        damOffspring = [];
                        damLevel++;
                        continue;
                    }

                    // Parse offspring data
                    if (currentDam && line.match(/^\d{2}\s+/)) {
                        const offspringMatch = line.match(/^(\d{2})\s+(.+?)\s+([SHV])\s+e\s+(.+)/);
                        if (offspringMatch) {
                            const year = parseInt(offspringMatch[1]);
                            const name = offspringMatch[2].trim();
                            const sex = offspringMatch[3];
                            const sirePart = offspringMatch[4];
                            
                            const earnings = parseEarnings(line);
                            const record = parseRaceRecord(line);
                            const time = parseTime(line);

                            const offspring = {
                                name,
                                year: year > 50 ? 1900 + year : 2000 + year,
                                sex,
                                sire: sirePart.split(/\s+/)[0],
                                earnings,
                                record,
                                time,
                                line: line
                            };
                            
                            damOffspring.push(offspring);
                        }
                    }
                }

                // Process final dam if exists
                if (currentDam && damOffspring.length > 0) {
                    processDamOffspring(horse, currentDam, damOffspring, damLevel - 1);
                }

                return horse;
            };

            const processDamOffspring = (horse, dam, offspring, level) => {
                if (!horse || !dam || !offspring.length) return;

                if (level === 0) {
                    // Process siblings (excluding main horse)
                    const siblings = offspring.filter(off => 
                        !off.name.toLowerCase().includes(horse.name.toLowerCase().split(' ')[0]) &&
                        !off.name.toLowerCase().includes(horse.name.toLowerCase().split(' ')[1] || 'xxxx')
                    );
                    
                    horse.siblingData = siblings;
                    horse.siblingEarnings = siblings.reduce((sum, s) => sum + s.earnings, 0);
                    horse.siblingStarts = siblings.reduce((sum, s) => sum + s.record.starts, 0);
                    horse.sibling1st = siblings.reduce((sum, s) => sum + s.record.wins, 0);
                    horse.sibling2nd = siblings.reduce((sum, s) => sum + s.record.seconds, 0);
                    horse.sibling3rd = siblings.reduce((sum, s) => sum + s.record.thirds, 0);
                    horse.siblingsStarted = siblings.filter(s => s.record.starts > 0).length;
                    horse.totalSiblings = siblings.length;
                    horse.siblingsBorn2020 = siblings.filter(s => s.year <= 2020).length;
                    
                } else if (level === 1) {
                    // Process granddam offspring (excluding dam)
                    const gdamOffspring = offspring.filter(off => 
                        !off.name.toLowerCase().includes(horse.dam.toLowerCase().split(' ')[0]) &&
                        !off.name.toLowerCase().includes(horse.dam.toLowerCase().split(' ')[1] || 'xxxx')
                    );
                    
                    horse.gdamEarnings = gdamOffspring.reduce((sum, s) => sum + s.earnings, 0);
                    horse.gdamStarts = gdamOffspring.reduce((sum, s) => sum + s.record.starts, 0);
                    horse.gdam1st = gdamOffspring.reduce((sum, s) => sum + s.record.wins, 0);
                    horse.gdam2nd = gdamOffspring.reduce((sum, s) => sum + s.record.seconds, 0);
                    horse.gdam3rd = gdamOffspring.reduce((sum, s) => sum + s.record.thirds, 0);
                    horse.gdamStarted = gdamOffspring.filter(s => s.record.starts > 0).length;
                    horse.gdamTotal = gdamOffspring.length;
                    horse.gdamBorn2020 = gdamOffspring.filter(s => s.year <= 2020).length;
                    
                } else if (level === 2) {
                    // Process great-granddam offspring (include all)
                    const ggdamOffspring = offspring;
                    
                    horse.ggdamEarnings = ggdamOffspring.reduce((sum, s) => sum + s.earnings, 0);
                    horse.ggdamStarts = ggdamOffspring.reduce((sum, s) => sum + s.record.starts, 0);
                    horse.ggdam1st = ggdamOffspring.reduce((sum, s) => sum + s.record.wins, 0);
                    horse.ggdam2nd = ggdamOffspring.reduce((sum, s) => sum + s.record.seconds, 0);
                    horse.ggdam3rd = ggdamOffspring.reduce((sum, s) => sum + s.record.thirds, 0);
                    horse.ggdamStarted = ggdamOffspring.filter(s => s.record.starts > 0).length;
                    horse.ggdamTotal = ggdamOffspring.length;
                    horse.ggdamBorn2020 = ggdamOffspring.filter(s => s.year <= 2020).length;
                }
            };

            const generateCSV = (data) => {
                const headers = [
                    'Horse', 'Page', 'Name', 'ID', 'Sex', 'Birth Date', 'Inbreeding %',
                    'Sire', 'Sire Earnings (SEK)', 'Sire Starts', 'Sire 1st', 'Sire 2nd', 'Sire 3rd', 'Sire Best Time',
                    'Dam', 'Dam Earnings (SEK)', 'Dam Starts', 'Dam 1st', 'Dam 2nd', 'Dam 3rd', 'Dam Birth Year', 'Dam BLUP', 'Dam Best Time',
                    'Dam\'s Dam', 'Granddam\'s Dam',
                    'Sibling Earnings (SEK)', 'Sibling Starts', 'Sibling 1st', 'Sibling 2nd', 'Sibling 3rd', 'Siblings Started', 'Total Siblings', 'Siblings Born ≤ 2020',
                    'Gdam Offspring Earnings (SEK, excl Dam)', 'Gdam Offspring Starts (excl Dam)', 'Gdam Offspring 1st (excl Dam)', 'Gdam Offspring 2nd (excl Dam)', 'Gdam Offspring 3rd (excl Dam)', 'Gdam Offspring Started (excl Dam)', 'Gdam Total Offspring (excl Dam)', 'Gdam Offspring Born ≤ 2020',
                    'Ggdam Offspring Earnings (SEK, incl Gdam)', 'Ggdam Offspring Starts (incl Gdam)', 'Ggdam Offspring 1st (incl Gdam)', 'Ggdam Offspring 2nd (incl Gdam)', 'Ggdam Offspring 3rd (incl Gdam)', 'Ggdam Offspring Started (incl Gdam)', 'Ggdam Total Offspring (incl Gdam)', 'Ggdam Offspring Born ≤ 2020'
                ];

                let csv = headers.join(';') + '\n';
                
                data.forEach(horse => {
                    const row = [
                        horse.name, horse.page, horse.name, horse.id, horse.sex, horse.birthDate, horse.inbreeding,
                        horse.sire, Math.round(horse.sireEarnings), horse.sireRecord.starts, horse.sireRecord.wins, horse.sireRecord.seconds, horse.sireRecord.thirds, horse.sireBestTime,
                        horse.dam, Math.round(horse.damEarnings), horse.damRecord.starts, horse.damRecord.wins, horse.damRecord.seconds, horse.damRecord.thirds, horse.damBirthYear || '', horse.damBLUP, horse.damBestTime,
                        horse.damsDam, horse.grandamsDam,
                        Math.round(horse.siblingEarnings), horse.siblingStarts, horse.sibling1st, horse.sibling2nd, horse.sibling3rd, horse.siblingsStarted, horse.totalSiblings, horse.siblingsBorn2020,
                        Math.round(horse.gdamEarnings), horse.gdamStarts, horse.gdam1st, horse.gdam2nd, horse.gdam3rd, horse.gdamStarted, horse.gdamTotal, horse.gdamBorn2020,
                        Math.round(horse.ggdamEarnings), horse.ggdamStarts, horse.ggdam1st, horse.ggdam2nd, horse.ggdam3rd, horse.ggdamStarted, horse.ggdamTotal, horse.ggdamBorn2020
                    ];
                    
                    csv += row.join(';') + '\n';
                });
                
                setCsvOutput(csv);
            };

            const downloadCSV = () => {
                if (!csvOutput) {
                    alert('No data to export. Please parse horse data first.');
                    return;
                }
                
                const horseName = parsedData[0]?.name.toLowerCase().replace(/\s+/g, '_') || 'horse';
                const blob = new Blob([csvOutput], { type: 'text/csv;charset=utf-8' });
                const url = window.URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = `${horseName}_pedigree_data.csv`;
                a.click();
                window.URL.revokeObjectURL(url);
            };

                setInputData('');
                setParsedData([]);
                setCsvOutput('');
                setActiveTab('input');
            };

            return React.createElement('div', { className: 'max-w-7xl mx-auto p-6 bg-gradient-to-br from-purple-50 to-blue-50 min-h-screen' },
                React.createElement('div', { className: 'bg-white rounded-xl shadow-2xl border border-purple-200' },

                        parsedData.length > 0 && React.createElement('div', { className: 'mt-4 p-4 bg-white/10 backdrop-blur rounded-lg border border-white/20' },
                            React.createElement('h3', { className: 'font-semibold text-white mb-2' }, 'Successfully Parsed:'),
                            React.createElement('div', { className: 'grid grid-cols-1 md:grid-cols-2 gap-2 text-sm text-purple-100' },
                                React.createElement('div', null, `Horse: ${parsedData[0]?.name}`),

                            React.createElement('button', { 
                                onClick: () => setActiveTab('input'),
                                className: 'mt-4 px-6 py-3 bg-purple-600 text-white rounded-lg hover:bg-purple-700'
                            }, 'Go to Input')
                        ),
                        
                        // CSV Export
                        csvOutput && React.createElement('div', { className: 'mt-8 pt-6 border-t border-gray-200' },

                                React.createElement('p', null, React.createElement('span', { className: 'font-semibold' }, 'Sire: '), horse.sire || 'Not found'),
                                React.createElement('p', null, React.createElement('span', { className: 'font-semibold' }, 'Dam: '), horse.dam || 'Not found')
                            ),
                            React.createElement('div', { className: 'space-y-3' },
                                React.createElement('h3', { className: 'font-semibold text-lg text-gray-700 mb-3' }, 'Performance Summary'),
                                React.createElement('p', { className: 'text-green-600 font-bold' },

                    'Gdam Offspring Earnings (SEK, excl Dam)', 'Gdam Offspring Starts (excl Dam)', 'Gdam Offspring 1st (excl Dam)', 'Gdam Offspring 2nd (excl Dam)', 'Gdam Offspring 3rd (excl Dam)', 'Gdam Offspring Started (excl Dam)', 'Gdam Total Offspring (excl Dam)', 'Gdam Offspring Born ≤ 2020',
                    'Ggdam Offspring Earnings (SEK, incl Gdam)', 'Ggdam Offspring Starts (incl Gdam)', 'Ggdam Offspring 1st (incl Gdam)', 'Ggdam Offspring 2nd (incl Gdam)', 'Ggdam Offspring 3rd (incl Gdam)', 'Ggdam Offspring Started (incl Gdam)', 'Ggdam Total Offspring (incl Gdam)', 'Ggdam Offspring Born ≤ 2020'
                ];

                let csv = headers.join(';') + '\n';
                
                data.forEach(horse => {
                    const row = [
                        horse.name, horse.page, horse.name, horse.id, horse.sex, horse.birthDate, horse.inbreeding,
                        horse.sire, Math.round(horse.sireEarnings), horse.sireRecord.starts, horse.sireRecord.wins, horse.sireRecord.seconds, horse.sireRecord.thirds, horse.sireBestTime,
                        horse.dam, Math.round(horse.damEarnings), horse.damRecord.starts, horse.damRecord.wins, horse.damRecord.seconds, horse.damRecord.thirds, horse.damBirthYear || '', horse.damBLUP, horse.damBestTime,
                        horse.damsDam, horse.grandamsDam,
                        Math.round(horse.siblingEarnings), horse.siblingStarts, horse.sibling1st, horse.sibling2nd, horse.sibling3rd, horse.siblingsStarted, horse.totalSiblings, horse.siblingsBorn2020,
                        Math.round(horse.gdamEarnings), horse.gdamStarts, horse.gdam1st, horse.gdam2nd, horse.gdam3rd, horse.gdamStarted, horse.gdamTotal, horse.gdamBorn2020,
                        Math.round(horse.ggdamEarnings), horse.ggdamStarts, horse.ggdam1st, horse.ggdam2nd, horse.ggdam3rd, horse.ggdamStarted, horse.ggdamTotal, horse.ggdamBorn2020
                    ];
                    
                    csv += row.join(';') + '\n';
                });
                
                setCsvOutput(csv);
            };

            const downloadCSV = () => {
                if (!csvOutput) {
                    alert('No data to export. Please parse horse data first.');
                    return;
                }
                
                const horseName = parsedData[0]?.name.toLowerCase().replace(/\s+/g, '_') || 'horse';
                const blob = new Blob([csvOutput], { type: 'text/csv;charset=utf-8' });
                const url = window.URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = `${horseName}_pedigree_data.csv`;
                a.click();
                window.URL.revokeObjectURL(url);
            };

            const clearData = () => {
                setInputData('');
                setParsedData([]);
                setCsvOutput('');
                setActiveTab('input');
            };

            return React.createElement('div', { className: 'max-w-7xl mx-auto p-6 bg-gradient-to-br from-purple-50 to-blue-50 min-h-screen' },
                React.createElement('div', { className: 'bg-white rounded-xl shadow-2xl border border-purple-200' },
                    // Header
                    React.createElement('div', { className: 'p-6 bg-gradient-to-r from-purple-600 to-blue-600 text-white rounded-t-xl' },
                        React.createElement('h1', { className: 'text-3xl font-bold mb-2' }, 'Universal Swedish Horse Pedigree Parser'),
                        React.createElement('p', { className: 'text-purple-100 text-lg' }, 'Automatically parses Swedish harness racing pedigree data'),
                        parsedData.length > 0 && React.createElement('div', { className: 'mt-4 p-4 bg-white/10 backdrop-blur rounded-lg border border-white/20' },
                            React.createElement('h3', { className: 'font-semibold text-white mb-2' }, 'Successfully Parsed:'),
                            React.createElement('div', { className: 'grid grid-cols-1 md:grid-cols-2 gap-2 text-sm text-purple-100' },
                                React.createElement('div', null, `Horse: ${parsedData[0]?.name}`),
                                React.createElement('div', null, `Sire: ${parsedData[0]?.sire || 'Not found'}`),
                                React.createElement('div', null, `Dam: ${parsedData[0]?.dam || 'Not found'}`),
                                React.createElement('div', null, 'Complete family performance data'),
                                React.createElement('div', null, `Total Family Earnings: ${parsedData[0]?.totalFamilyEarnings.toLocaleString()} SEK`),
                                React.createElement('div', null, `Siblings: ${parsedData[0]?.siblingData.length || 0} found`)
                            )
                        )
                    ),
                    
                    // Navigation Tabs
                    React.createElement('div', { className: 'flex border-b border-gray-200' },
                        ['input', 'overview', 'pedigree', 'family'].map(tab =>
                            React.createElement('button', {
                                key: tab,
                                onClick: () => setActiveTab(tab),
                                className: `px-6 py-3 font-medium capitalize transition-colors ${
                                    activeTab === tab 
                                        ? 'border-b-2 border-purple-500 text-purple-600 bg-purple-50' 
                                        : 'text-gray-600 hover:text-gray-800 hover:bg-gray-50'
                                }`
                            }, 
                            tab === 'input' ? 'Input Data' : 
                            tab === 'overview' ? 'Overview' : 
                            tab === 'pedigree' ? 'Pedigree' : 'Family')
                        )
                    ),
                    
                    // Content
                    React.createElement('div', { className: 'p-6' },
                        activeTab === 'input' ? renderInput() :
                        parsedData.length > 0 ? (
                            activeTab === 'overview' ? renderOverview(parsedData[0]) :
                            activeTab === 'pedigree' ? renderPedigree(parsedData[0]) :
                            renderFamily(parsedData[0])
                        ) : React.createElement('div', { className: 'text-center py-12' },
                            React.createElement('div', { className: 'text-gray-400 text-6xl mb-4' }, '🐎'),
                            React.createElement('p', { className: 'text-gray-500 italic text-lg' }, 'Please input horse data first'),
                            React.createElement('button', { 
                                onClick: () => setActiveTab('input'),
                                className: 'mt-4 px-6 py-3 bg-purple-600 text-white rounded-lg hover:bg-purple-700'
                            }, 'Go to Input')
                        ),
                        
                        // CSV Export
                        csvOutput && React.createElement('div', { className: 'mt-8 pt-6 border-t border-gray-200' },
                            React.createElement('div', { className: 'flex items-center justify-between mb-4' },
                                React.createElement('h3', { className: 'text-lg font-semibold text-gray-800' }, 'Export Data'),
                                React.createElement('div', { className: 'space-x-3' },
                                    React.createElement('button', {
                                        onClick: downloadCSV,
                                        className: 'px-6 py-3 bg-green-600 text-white rounded-lg hover:bg-green-700 font-medium'
                                    }, 'Download CSV'),
                                    React.createElement('button', {
                                        onClick: clearData,
                                        className: 'px-6 py-3 bg-red-600 text-white rounded-lg hover:bg-red-700 font-medium'
                                    }, 'Clear All')
                                )
                            ),
                            React.createElement('div', { className: 'bg-gray-50 rounded-lg p-4' },
                                React.createElement('textarea', {
                                    value: csvOutput.split('\n').slice(0, 3).join('\n') + '\n...',
                                    readOnly: true,
                                    className: 'w-full h-24 p-3 border rounded font-mono text-xs bg-white resize-none'
                                }),
                                React.createElement('p', { className: 'text-sm text-gray-600 mt-2' }, 
                                    `Ready for export: ${csvOutput.split('\n').length - 2} data rows`
                                )
                            )
                        )
                    )
                )
            );

            function renderInput() {
                return React.createElement('div', { className: 'space-y-6' },
                    React.createElement('div', { className: 'bg-blue-50 rounded-lg p-6 border border-blue-200' },
                        React.createElement('h2', { className: 'text-2xl font-bold text-blue-800 mb-4' }, 'Input Horse Pedigree Data'),
                        React.createElement('p', { className: 'text-blue-700 mb-4' }, 
                            'Paste your MEMORY SISU pedigree data or any other Swedish harness racing pedigree data below. The parser will automatically detect the horse information, sire/dam details, and family performance data.'
                        ),
                        React.createElement('textarea', {
                            value: inputData,
                            onChange: (e) => setInputData(e.target.value),
                            placeholder: 'Paste your horse pedigree data here...\n\nFor example:\n115 MEMORY SISU\n22-1288 br sto f. 22 04 16\n\nDjango Riff                Ready Cash...',
                            className: 'w-full h-96 p-4 border border-gray-300 rounded-lg font-mono text-sm resize-none',
                            spellCheck: false
                        }),
                        React.createElement('div', { className: 'flex items-center justify-between mt-4' },
                            React.createElement('p', { className: 'text-sm text-gray-600' }, 
                                `Characters: ${inputData.length}`
                            ),
                            React.createElement('div', { className: 'space-x-3' },
                                React.createElement('button', {
                                    onClick: () => setInputData(''),
                                    className: 'px-4 py-2 bg-gray-500 text-white rounded hover:bg-gray-600'
                                }, 'Clear'),
                                React.createElement('button', {
                                    onClick: parseHorseData,
                                    disabled: !inputData.trim(),
                                    className: 'px-6 py-2 bg-purple-600 text-white rounded hover:bg-purple-700 disabled:bg-gray-300 disabled:cursor-not-allowed font-medium'
                                }, 'Parse Data')
                            )
                        )
                    )
                );
            }

            function renderOverview(horse) {
                return React.createElement('div', { className: 'space-y-6' },
                    React.createElement('div', { className: 'bg-gradient-to-r from-purple-100 to-blue-100 rounded-lg p-6 border border-purple-200' },
                        React.createElement('h2', { className: 'text-2xl font-bold text-purple-800 mb-4' }, `${horse.page} - ${horse.name}`),
                        React.createElement('div', { className: 'grid grid-cols-1 md:grid-cols-3 gap-6' },
                            React.createElement('div', { className: 'space-y-3' },
                                React.createElement('h3', { className: 'font-semibold text-lg text-gray-700 mb-3' }, 'Horse Details'),
                                React.createElement('p', null, React.createElement('span', { className: 'font-semibold' }, 'ID: '), horse.id),
                                React.createElement('p', null, React.createElement('span', { className: 'font-semibold' }, 'Sex: '), horse.sex),
                                React.createElement('p', null, React.createElement('span', { className: 'font-semibold' }, 'Birth: '), horse.birthDate),
                                React.createElement('p', null, React.createElement('span', { className: 'font-semibold' }, 'Sire: '), horse.sire || 'Not found'),
                                React.createElement('p', null, React.createElement('span', { className: 'font-semibold' }, 'Dam: '), horse.dam || 'Not found')
                            ),
                            React.createElement('div', { className: 'space-y-3' },
                                React.createElement('h3', { className: 'font-semibold text-lg text-gray-700 mb-3' }, 'Performance Summary'),
                                React.createElement('p', { className: 'text-green-600 font-bold' }, 
                                    React.createElement('span', { className: 'font-semibold' }, 'Total Family Earnings: '), 
                                    `${horse.totalFamilyEarnings.toLocaleString()} SEK`
                                ),
                                React.createElement('p', null, React.createElement('span', { className: 'font-semibold' }, 'Total Family Starts: '), horse.totalFamilyStarts),
                                React.createElement('p', null, React.createElement('span', { className: 'font-semibold' }, 'Siblings Found: '), horse.siblingData.length),
                                React.createElement('p', { className: 'text-purple-600 font-semibold' }, 
                                    React.createElement('span', { className: 'font-semibold' }, 'Sire Earnings: '), 
                                    `${horse.sireEarnings.toLocaleString()} SEK`
                                ),
                                React.createElement('p', { className: 'text-blue-600 font-semibold' }, 
                                    React.createElement('span', { className: 'font-semibold' }, 'Dam Earnings: '), 
                                    `${horse.damEarnings.toLocaleString()} SEK`
                                )
                            ),
                            React.createElement('div', { className: 'space-y-3' },
                                React.createElement('h3', { className: 'font-semibold text-lg text-gray-700 mb-3' }, 'Race Records'),
                                React.createElement('div', { className: 'space-y-2' },
                                    React.createElement('p', { className: 'text-sm' }, 
                                        React.createElement('span', { className: 'font-semibold' }, 'Sire Record: '), 
                                        `${horse.sireRecord.starts} starts (${horse.sireRecord.wins}-${horse.sireRecord.seconds}-${horse.sireRecord.thirds})`
                                    ),
                                    React.createElement('p', { className: 'text-sm' }, 
                                        React.createElement('span', { className: 'font-semibold' }, 'Dam Record: '), 
                                        `${horse.damRecord.starts} starts (${horse.damRecord.wins}-${horse.damRecord.seconds}-${horse.damRecord.thirds})`
                                    ),
                                    React.createElement('p', { className: 'text-sm' }, 
                                        React.createElement('span', { className: 'font-semibold' }, 'Sire Best Time: '), 
                                        horse.sireBestTime || 'Not recorded'
                                    ),
                                    React.createElement('p', { className: 'text-sm' }, 
                                        React.createElement('span', { className: 'font-semibold' }, 'Dam Best Time: '), 
                                        horse.damBestTime || 'Not recorded'
                                    )
                                )
                            )
                        )
                    )
                );
            }

            function renderPedigree(horse) {
                return React.createElement('div', { className: 'space-y-6' },
                    React.createElement('h2', { className: 'text-2xl font-bold text-gray-800 mb-6' }, 'Pedigree Structure'),
                    React.createElement('div', { className: 'bg-gray-50 rounded-lg p-6 border' },
                        React.createElement('div', { className: 'text-center space-y-4' },
                            React.createElement('div', { className: 'bg-purple-600 text-white p-4 rounded-lg font-bold text-xl' }, horse.name),
                            React.createElement('div', { className: 'grid grid-cols-2 gap-4 mt-4' },
                                React.createElement('div', { className: 'bg-blue-100 p-4 rounded-lg border-2 border-blue-300' },
                                    React.createElement('h3', { className: 'font-bold text-blue-800 mb-2' }, 'Sire'),
                                    React.createElement('p', { className: 'font-semibold text-lg' }, horse.sire || 'Unknown'),
                                    React.createElement('p', { className: 'text-sm text-blue-600' }, `${horse.sireEarnings.toLocaleString()} SEK`),
                                    React.createElement('p', { className: 'text-sm text-blue-600' }, horse.sireBestTime || 'No time recorded'),
                                    React.createElement('p', { className: 'text-xs text-blue-500' }, 
                                        `${horse.sireRecord.starts} starts (${horse.sireRecord.wins}-${horse.sireRecord.seconds}-${horse.sireRecord.thirds})`
                                    )
                                ),
                                React.createElement('div', { className: 'bg-pink-100 p-4 rounded-lg border-2 border-pink-300' },
                                    React.createElement('h3', { className: 'font-bold text-pink-800 mb-2' }, 'Dam'),
                                    React.createElement('p', { className: 'font-semibold text-lg' }, horse.dam || 'Unknown'),
                                    React.createElement('p', { className: 'text-sm text-pink-600' }, `${horse.damEarnings.toLocaleString()} SEK`),
                                    React.createElement('p', { className: 'text-sm text-pink-600' }, horse.damBestTime || 'No time recorded'),
                                    React.createElement('p', { className: 'text-xs text-pink-500' }, 
                                        `${horse.damRecord.starts} starts (${horse.damRecord.wins}-${horse.damRecord.seconds}-${horse.damRecord.thirds})`
                                    )
                                )
                            ),
                            React.createElement('div', { className: 'grid grid-cols-2 gap-4 mt-4' },
                                React.createElement('div', { className: 'bg-yellow-100 p-3 rounded border' },
                                    React.createElement('h4', { className: 'font-semibold text-yellow-800' }, 'Dam\'s Dam'),
                                    React.createElement('p', { className: 'text-sm' }, horse.damsDam || 'Unknown')
                                ),
                                React.createElement('div', { className: 'bg-green-100 p-3 rounded border' },
                                    React.createElement('h4', { className: 'font-semibold text-green-800' }, 'Granddam\'s Dam'),
                                    React.createElement('p', { className: 'text-sm' }, horse.grandamsDam || 'Unknown')
                                )
                            )
                        )
                    )
                );
            }

            function renderFamily(horse) {
                const siblingStats = horse.siblingData.reduce((acc, sibling) => ({
                    earnings: acc.earnings + sibling.earnings,
                    starts: acc.starts + sibling.record.starts,
                    wins: acc.wins + sibling.record.wins
                }), { earnings: 0, starts: 0, wins: 0 });

                return React.createElement('div', { className: 'space-y-6' },
                    React.createElement('h2', { className: 'text-2xl font-bold text-gray-800 mb-6' }, 'Family Performance'),
                    
                    React.createElement('div', { className: 'bg-green-50 rounded-lg p-6 border border-green-200' },
                        React.createElement('h3', { className: 'text-xl font-semibold text-green-800 mb-4' }, `Siblings (${horse.siblingData.length} found)`),
                        React.createElement('div', { className: 'grid grid-cols-1 md:grid-cols-3 gap-4 mb-4' },
                            React.createElement('div', { className: 'text-center p-3 bg-white rounded border' },
                                React.createElement('p', { className: 'text-2xl font-bold text-green-600' }, siblingStats.earnings.toLocaleString()),
                                React.createElement('p', { className: 'text-sm text-gray-600' }, 'Total Sibling Earnings (SEK)')
                            ),
                            React.createElement('div', { className: 'text-center p-3 bg-white rounded border' },
                                React.createElement('p', { className: 'text-2xl font-bold text-blue-600' }, siblingStats.starts),
                                React.createElement('p', { className: 'text-sm text-gray-600' }, 'Total Sibling Starts')
                            ),
                            React.createElement('div', { className: 'text-center p-3 bg-white rounded border' },
                                React.createElement('p', { className: 'text-2xl font-bold text-purple-600' }, siblingStats.wins),
                                React.createElement('p', { className: 'text-sm text-gray-600' }, 'Total Sibling Wins')
                            )
                        ),
                        
                        React.createElement('div', { className: 'space-y-2' },
                            React.createElement('h4', { className: 'font-semibold text-gray-700 mb-2' }, 'Individual Siblings:'),
                            horse.siblingData.length > 0 ? horse.siblingData.map((sibling, index) =>
                                React.createElement('div', { key: index, className: 'bg-white p-3 rounded border flex justify-between items-center' },
                                    React.createElement('div', null,
                                        React.createElement('span', { className: 'font-semibold' }, sibling.name),
                                        React.createElement('span', { className: 'text-gray-500 ml-2' }, `(${sibling.year}, ${sibling.sex})`),
                                        React.createElement('span', { className: 'text-gray-500 ml-2' }, `by ${sibling.sire}`)
                                    ),
                                    React.createElement('div', { className: 'text-right text-sm' },
                                        React.createElement('p', { className: 'font-semibold' }, `${sibling.earnings.toLocaleString()} SEK`),
                                        React.createElement('p', { className: 'text-gray-500' }, 
                                            `${sibling.record.starts} starts (${sibling.record.wins}-${sibling.record.seconds}-${sibling.record.thirds})`
                                        ),
                                        sibling.time && React.createElement('p', { className: 'text-gray-500' }, sibling.time)
                                    )
                                )
                            ) : React.createElement('p', { className: 'text-gray-500 italic' }, 'No sibling data found')
                        )
                    )
                );
            }
        };

        ReactDOM.render(React.createElement(UniversalHorseParser), document.getElementById('root'));
    </script>
